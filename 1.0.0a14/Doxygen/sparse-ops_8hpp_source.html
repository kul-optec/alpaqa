23/// Returns a range over the row indices in the given column of @p sp_mat that
24/// are also in @p mask.
25/// The range consists of the full Eigen InnerIterators (row, column, value).
26template <class SpMat, class MaskVec>
27auto select_rows_in_col(const SpMat &sp_mat, MaskVec &mask, auto column) {
28 // Make a range that iterates over all matrix elements in the given column:
29 using row_iter_t = typename SpMat::InnerIterator;
30 util::iter_range_adapter<row_iter_t> col_range{{sp_mat, column}};
31 // Projector that extracts the row index from an element of that range:
32 static constexpr auto proj_row = [](const row_iter_t &it) {
33 return static_cast<typename MaskVec::value_type>(it.row());
34 };
35 // Compute the intersection between the matrix elements and the mask:
36 auto intersection = util::iter_set_intersection(std::move(col_range),
37 std::ranges::ref_view{mask},
38 std::less{}, proj_row);
39 // Extract just the iterator to the matrix element (dropping the mask):
40 auto extract_eigen_iter = []<class T>(T &&tup) -> decltype(auto) {
41 return std::get<0>(std::forward<T>(tup));
42 };
43 return std::views::transform(std::move(intersection), extract_eigen_iter);
44}

46/// Like @ref select_rows_in_col, but returns a range of tuples containing the
47/// Eigen InnerIterator and a linear index into the mask.
48template <class SpMat, class MaskVec>
49auto select_rows_in_col_iota(const SpMat &sp_mat, MaskVec &mask, auto column) {
50 // Make a range that iterates over all matrix elements in the given column:
51 using row_iter_t = typename SpMat::InnerIterator;
52 util::iter_range_adapter<row_iter_t> col_range{{sp_mat, column}};
53 // Projector that extracts the row index from an element of that range:
54 static constexpr auto proj_row = [](const row_iter_t &it) {
55 return static_cast<typename MaskVec::value_type>(it.row());
56 };
57 // Make a range of tuples of the index into the mask and the mask value:
58 auto iota_mask = util::enumerate(std::ranges::ref_view{mask});
59 // Projector that extracts the mask value from such a tuple:
60 static constexpr auto proj_mask = [](const auto &tup) -> decltype(auto) {
61 return std::get<1>(tup);
62 };
63 // Compute the intersection between the matrix elements and the mask:
64 auto intersection =
65 util::iter_set_intersection(std::move(col_range), std::move(iota_mask),
66 std::less{}, proj_row, proj_mask);
67 // Extract the iterator to the matrix element and the index into the mask:
68 auto extract_eigen_iter_and_index = []<class T>(T && tup)
69 requires(std::is_rvalue_reference_v<T &&>)
70 {
71 auto &[eigen_iter, enum_tup] = tup;
72 auto &mask_index = std::get<0>(enum_tup);
73 return std::tuple{std::move(eigen_iter), std::move(mask_index)};
74 };
75 return std::views::transform(std::move(intersection),
76 extract_eigen_iter_and_index);
77}

82template <class SpMat, class Mat, class MaskVec>
83void sparse_add_masked(const SpMat &R_full, Mat &&R, const MaskVec &mask) {
84 // Iterate over all columns in the mask
85 for (auto [ci, c] : util::enumerate(mask))

87 for (auto [r, ri] : detail::select_rows_in_col_iota(R_full, mask, c))

131template <class I>
132void convert_triplets_to_ccs(const auto &rows, const auto &cols,
133 auto &&inner_idx, auto &&outer_ptr, I idx_0 = 0) {
134 // Inner indices: simply the row indices
135 assert(std::size(rows) == std::size(inner_idx));
136 auto cvt_indices = [&](auto i) { return static_cast<I>(i) - idx_0; };
137 std::ranges::ref_view rows_vw = rows;
138 std::ranges::transform(rows_vw, std::begin(inner_idx), cvt_indices);
139 // Outer indices: need to count the number of nonzeros per column
140 auto cols_iter = std::begin(cols);
141 for (auto &&[i, outer] : std::views::enumerate(outer_ptr)) {
142 cols_iter = std::lower_bound(cols_iter, std::end(cols), i + idx_0);
143 outer = static_cast<I>(cols_iter - std::begin(cols));
144 }
145}

158/// Sort the (row, column, value) triplets, by column index first, then row.
159template <class... Ts>
160void sort_triplets(Ts &&...triplets) {
161 // Sort the indices (by column first, then row)
162 auto cmp = [](const auto &a, const auto &b) {
163 return std::tie(std::get<1>(a), std::get<0>(a)) <
164 std::tie(std::get<1>(b), std::get<0>(b));

179/// Given a sparse compressed-column storage matrix, sort all row indices
180/// within each column.
181template <class Outer, class... Inners>
182void sort_rows_csc(const Outer &outer_ptr, Inners &&...inners) {
183 if (outer_ptr.size() == 0)
184 return;
185 // Sort the indices (by row)
186 auto cmp = [](const auto &a, const auto &b) {
187 return std::get<0>(a) < std::get<0>(b);
188 };
189 for (decltype(outer_ptr.size()) c = 0; c < outer_ptr.size() - 1; ++c) {
190 auto inner_start = outer_ptr(c);
191 auto inner_end = outer_ptr(c + 1);
192 auto indices = std::views::zip(
193 std::ranges::subrange{std::ranges::begin(inners) + inner_start,
194 std::ranges::begin(inners) + inner_end}...);
195 std::ranges::sort(indices, cmp);
196 }
197}

208/// Check that no two entries with the same row and column index exist in
209/// the given sparse compressed-column storage matrix. Assumes sorted indices.
210template <class Outer, class Inner>
211bool check_uniqueness_csc(const Outer &outer_ptr, const Inner inner) {
212 if (outer_ptr.size() == 0)
213 return true;
214 auto is_unique_col = [&](auto &inner_start) {
215 auto inner_end = (&inner_start)[1];
216 auto indices =
217 std::ranges::subrange{std::ranges::begin(inner) + inner_start,
218 std::ranges::begin(inner) + inner_end};
219 return std::ranges::adjacent_find(indices, std::equal_to<>{}) ==
220 std::ranges::end(indices);
221 };
222 return std::transform_reduce(std::begin(outer_ptr), std::end(outer_ptr) - 1,
223 true, std::logical_and<>{}, is_unique_col);
224}
